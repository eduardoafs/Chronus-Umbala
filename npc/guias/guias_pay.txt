			set @COMPASS_CHECK,1;

		next;
	sM_Menu:
		menu 	"^FF0000Guilda dos Arquiros^000000",M_1_1, "Loja de Armas",M_1_2, "Loja de Ferramentas",M_1_3, "Estalagaem",M_1_4, "Palácio Central",M_1_5,
			"Cancelar",-;
	M_Limpar:
		show "[Guia de Payon]";
		show "Se você deseja apagar as indicaçoes do seu mini-mapa, por favor selecione 'Limpar Indicações'.";
		next;
		goto M_Menu;

   M_1_1: 
         show "[Guia de Payon]"; 
         show "A Guilda dos "+callfunc("classname",3,1,1)+" é o lugal onde "+callfunc("classname",0,1,1)+" pode se tornar "+callfunc("classname",3,1,1)+".  Ela se encontra ao Norte da Vila dos "+callfunc("classname",3,1,1)+". Para chegar na vila dos "+callfunc("classname",3,1,1)+", saia da cidade pela direção nordeste."; 
         viewpoint 1,175,189,1,0xFF0000; 
         goto sL_Wipe; 
      M_1_2: 
         show "[Guia de Payon]"; 
         show "A 'Loja de Armas' se encontra ao noroeste da cidade."; 
         viewpoint 1,30,177,2,0xFF00FF; 
         goto sL_Wipe; 
      M_1_3: 
         show "[Guia de Payon]"; 
         show "A 'Loja de Ferramentas ' se encontra ao noroeste da Vila dos "+callfunc("classname",3,1,1)+". Para chegar na vila dos "+callfunc("classname",3,1,1)+", saia da cidade pela direção nordeste.";
         viewpoint 1,175,189,3,0xFF00FF; 
         goto sL_Wipe; 
      M_1_4: 
         show "[Guia de Payon]"; 
         show "A 'Estalagem' se encotra a leste do centro da cidade.";
         viewpoint 1,172,133,4,0xFF00FF; 
         goto sL_Wipe; 
      M_1_5: 
         show "[Guia de Payon]"; 
         show "O 'Palácio Central' se encontra ao norte do centro da cidade."; 
         viewpoint 1,90,167,5,0x00FF00; 
         goto sL_Wipe; 
      M_1_6: 
         set @COMPASS_CHECK, 2; 

      sL_Wipe: 
         if(@COMPASS_CHECK != 2) next;   // desvia o segundo 'prox' quando 'limpar indicações'é escolhido
         if(@COMPASS_CHECK == 1) goto sM_Menu; 
         viewpoint 2,175,189,1,0xFF0000; 
         viewpoint 2,30,177,2,0xFF00FF; 
         viewpoint 2,175,189,3,0xFF00FF; 
         viewpoint 2,172,133,4,0xFF00FF; 
         viewpoint 2,90,167,5,0x00FF00; 
         if (@COMPASS_CHECK == 0) goto sM_Menu; 
         goto M_Menu;

	M_2:
		show "[Guia de Payon]";
		show "Nós fomos exaustivamente treinados para serví-lo da melhor possivel e sem falhas. Assim você tem a garantia das melhores indicações possíveis.";
		next;
		show "[Guia de Payon]";
		show "As indicações que lhe daremos podem ser marcadas no seu mini-mapa	.";
		show "Para visualizar o seu mini-mapa, aperte 'ctrl+tab' ou clique no botão 'mapa' no menu principal.";
		show "Controle o seu 'zoom' no mini-mapa por meio dos símbolos '+' e '-'.";
		goto M_Menu;

	M_End:
		show "[Guia de Payon]";
		show "Espero que você tenha uma viagem segura e aproveite sua estadia.";
		close;
}
