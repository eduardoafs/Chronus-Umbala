morocc.gat,148,86,5	script	Mercenário Shifty	725,{
set $ninja_price,250000;

// INICIAR O MENU //
M_Start:
show "[Mercenário Shifty]";
show "O que você deseja?";
next;
if (getgmlevel(3) > 90) goto M_GM;
menu "Comprar Ninjas",M_Buy,"Assassinar alguém",M_Kill,"Verificar seus Ninjas",M_Check,"Cancelar",M_Exit;
M_GM:
menu "Comprar Ninjas",M_Buy,"Assassinar alguém",M_Kill,"Verificar seus Ninjas",M_Check,"Adicionar Ninjas",M_Add,"Cancelar",M_Exit;

// MENU DO GM PARA ADICIONAR NINJAS //
M_Add:
show "[Mercenário Shifty]";
show "Quantos ninjas disponíveis você deseja fazer?";
next;
set @add,0;
digit @add;
set $ninja_avail,$ninja_avail+@add;
show @add + " ninjas adicionado.";
close;

// Comprar NINJAS //
M_Buy:
show "[Mercenário Shifty]";
show "Quantos ninjas você deseja compra?";
show "Há ^0000FF" + $ninja_avail + "^000000 ninjas disponíveis.";
show "Custa ^0000FF" + $ninja_price + " zeny ^000000cada um.";

set @buy,0;
digit @buy;
next;
if ($ninja_avail < 1) goto NoNinjas;
if ($ninja_avail < @buy) goto NotEnoughNinjas;
set @price,@buy*$ninja_price;
if (zeny < @price ) goto NoZeny;

show "[Mercenário Shifty]";
show "Isso custará a você  ^0000FF" + @price + " zeny^000000.";
next;
menu "Continue",-,"Cancelar",M_Exit;

set zeny,zeny-@price;
set #ninjas,#ninjas+@buy;
set $ninja_avail,$ninja_avail-@buy;

show "[Mercenário Shifty]";
show "Obrigado.";
close;

// ASSASSINAR ALGUÉM //
M_Kill:
if ($AgitStarted != 0) goto M_Busy;
show "[Mercenário Shifty]";
show "Ponha o nome do alvo.";
show "^FF0000Datilografe o nome exatamente, se não, eu não poderei encontrar a vítima.^000000";
next;
menu "Continue",-,"Cancelar",M_Exit;
set @name$,"0";
digit @name$;
next;
show "[Mercenário Shifty]";
show "Ninjas Ativos: "+#ninjas;
show "Ninjas Descansando: "+#ninjasr;
show "Quanto você quer emitir?";
set @number,0;
digit @number;
if (@number < 1) goto NoNinjasSent;
if (@number > #ninjas) goto NotEnoughNinjas1;
if (@number > 10) goto TooManyNinjas;
set @chance,rand (1,12);
set #ninjas,#ninjas-@number;
set #ninjas,#ninjas+#ninjasr;
set #ninjasr,0;
if (@number < @chance) goto M_Failure;

// ATAQUE BEM SUCEDIDO //
show "Emitindo ninjas agora.";
next;
show "[Mercenário Shifty]";
set @ninjasurvived,rand (1,@number);
set #ninjasr,@number-@ninjasurvived;
show "Seu ataque foi bem sucedido, mas somente ^FF0000" + #ninjasr + "^000000 Ninjas sobreviveu.";

atcommand strcharinfo(0) + "@kill "+@name$;
announce @name$+" foi assassinado por " + strcharinfo(0) +"'s Ninjas.",8;
close;

// ATAQUE FALHOU //
M_Failure:
show "Emitindo ninjas agora.";
next;
show "[Mercenário Shifty]";
set @ninjasurvived,rand (1,@number);
set #ninjasr,@number-@ninjasurvived;
show "Seu ataque falhOU e somente ^FF0000" + #ninjasr + "^000000 Ninjas sobreviveu.";

announce @name$+" sobreviveu " + strcharinfo(0) +"'s aos ataque de Ninja.",8;
close;

// NINJAS OCUPADO PARA O WOE //
M_Busy:
show "[Mercenário Shifty]";
show "Desculpe, todos meus ninjas estão ocupados para fazer a guerra de Emperium.";
close;

// VERIFIQUE SEUS NINJAS //
M_Check:
show "[Mercenário Shifty]";
show "Você tem:";
show "^FF0000" + #ninjas + "^000000 Ninjas Ativos.";
show "^0000FF" + #ninjasr + "^000000 Ninjas Descansando.";
next;
goto M_Start;


// LIMITE //
NoNinjasSent:
show "[Mercenário Shifty]";
show "Você não pode matar qualquer um sem ninjas.";
next;
goto M_Start;

TooManyNinjas:
show "[Mercenário Shifty]";
show "Você só pode emitir 10 ninjas no máximo.";
next;
goto M_Start;

NoZeny:
show "[Mercenário Shifty]";
show "Você não tem bastante zeny.";
close;

NotEnoughNinjas:
show "[Mercenário Shifty]";
show "Não há muitos ninjas para comprar.";
next;
goto M_Start;

NoNinjas:
show "[Mercenário Shifty]";
show "Não há nenhum ninjas a comprar.";
close;

NotEnoughNinjas1:
show "[Mercenário Shifty]";
show "Você não tem muitos ninjas.";
next;
goto M_Start;

M_Exit:
show "[Mercenário Shifty]";
show "Adeus.";
close;

// TEMPO DE ATRASO PARA ADICIONAR NINJA

OnClock0600:
set $ninja_avail,$ninja_avail+2;
end;

OnClock1200:
set $ninja_avail,$ninja_avail+2;
end;

OnClock1500:
set $ninja_avail,$ninja_avail+2;
end;


OnClock1800:
set $ninja_avail,$ninja_avail+3;
end;

OnClock1900:
set $ninja_avail,$ninja_avail+2;
end;

OnClock2000:
set $ninja_avail,$ninja_avail+2;
end;


OnClock0000:
set $ninja_avail,$ninja_avail+2;
end;

OnInit:
set $ninja_avail,$ninja_avail+1;
end;
}
